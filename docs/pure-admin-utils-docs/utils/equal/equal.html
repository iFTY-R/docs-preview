isEqualObject ​

判断两个对象是否相等

基础用法

​

{"a":"test"} 等于 {"a":"test"}
{"a":"test"} 不等于 {"a":"test1"}
查看代码
vue
<script setup lang="ts">
import { ref } from "vue";
import { isEqualObject } from "@pureadmin/utils";

let obj1 = ref({ a: "test" });
let obj2 = ref({ a: "test" });
let obj3 = ref({ a: "test1" });
</script>

<template>
  <naive-theme>
    <n-space className="mt-2" vertical>
      <n-gradient-text type="info">
        {{
          isEqualObject(obj1, obj2)
            ? `${JSON.stringify(obj1)} 等于 ${JSON.stringify(obj2)}`
            : `${JSON.stringify(obj1)} 不等于 ${JSON.stringify(obj2)}`
        }}
      </n-gradient-text>

      <n-gradient-text type="info">
        {{
          isEqualObject(obj1, obj3)
            ? `${JSON.stringify(obj1)} 等于 ${JSON.stringify(obj3)}`
            : `${JSON.stringify(obj1)} 不等于 ${JSON.stringify(obj3)}`
        }}
      </n-gradient-text>
    </n-space>
  </naive-theme>
</template>

isEqualArray ​

判断两个数组是否相等

基础用法

​

[1,2] 等于 [1,2]
[1,2] 不等于 [1,2,3]
查看代码
vue
<script setup lang="ts">
import { ref } from "vue";
import { isEqualArray } from "@pureadmin/utils";

let array1 = ref([1, 2]);
let array2 = ref([1, 2]);
let array3 = ref([1, 2, 3]);
</script>

<template>
  <naive-theme>
    <n-space className="mt-2" vertical>
      <n-gradient-text type="info">
        {{
          isEqualArray(array1, array2)
            ? `[${array1}] 等于 [${array2}]`
            : `[${array1}] 不等于 [${array2}]`
        }}
      </n-gradient-text>

      <n-gradient-text type="info">
        {{
          isEqualArray(array1, array3)
            ? `[${array1}] 等于 [${array3}]`
            : `[${array1}] 不等于 [${array3}]`
        }}
      </n-gradient-text>
    </n-space>
  </naive-theme>
</template>
